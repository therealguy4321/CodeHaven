import tkinter as tk
from tkinter import ttk

def on_submit():
    selected_month = month_var.get()
    selected_day = day_var.get()
    result_label.config(text=f"Selected Month: {selected_month}, Selected Day: {selected_day}")

# Create main window
root = tk.Tk()
root.title("Month and Day Selection")

# Month dropdown
months = ["January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"]
month_label = ttk.Label(root, text="Select Month:")
month_label.grid(row=0, column=0, padx=10, pady=10)
month_var = tk.StringVar()
month_dropdown = ttk.Combobox(root, textvariable=month_var, values=months)
month_dropdown.grid(row=0, column=1, padx=10, pady=10)
month_dropdown.set(months[0])  # Set the default value

# Day dropdown
days = list(range(1, 32))
day_label = ttk.Label(root, text="Select Day:")
day_label.grid(row=1, column=0, padx=10, pady=10)
day_var = tk.StringVar()
day_dropdown = ttk.Combobox(root, textvariable=day_var, values=days)
day_dropdown.grid(row=1, column=1, padx=10, pady=10)
day_dropdown.set(days[0])  # Set the default value

# Submit button
submit_button = ttk.Button(root, text="Submit", command=on_submit)
submit_button.grid(row=2, column=0, columnspan=2, pady=10)

# Result label
result_label = ttk.Label(root, text="")
result_label.grid(row=3, column=0, columnspan=2, pady=10)

# Run the GUI
root.mainloop()
